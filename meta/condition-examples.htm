General 4.5.3.1

General Condition Example

General Condition Example (id = "example")

Raw XML

<Condition xmlns="http://hl7.org/fhir">
  <text>
   <status value="generated"/>
   <div xmlns="http://www.w3.org/1999/xhtml">Severe burn of left ear (Date: 24-May 2012)</div>   
 </text>
  <subject>
    <type value="Patient"/>
    <reference value="patient/@example"/>
  </subject>
  <code> 
    <coding>
      <system value="http://snomed.info/id"/>
      <code value="39065001"/>
      <display value="Burn of ear"/>
    </coding>
    <text value="Burnt Ear"/>
  </code>
 <category>
   <coding>
    <system value="http://hl7.org/fhir/condition-category"/>
    <code value="diagnosis"/> 
    <display value="Diagnosis"/>
   </coding>
   <!--   and also a snomed coding   -->
   <coding>
    <system value="http://snomed.info/id"/>
    <code value="439401001"/> 
    <display value="Diagnosis"/>
   </coding>
 </category>
 <status value="confirmed"/>
 <severity>
   <coding>
    <system value="http://snomed.info/id"/>
    <code value="24484000"/>
    <display value="Severe"/>
   </coding>
 </severity>
 <onsetDate value="2012-05-24"/>
 <location>
  <code>
    <coding>
      <system value="http://snomed.info/id"/>
      <code value="49521004"/>
      <display value="Left external ear structure"/>
    </coding>
    <text value="Left Ear"/>
  </code>
 </location>
</Condition>
